\subsection{Physics-Informed Hybrid Search}\label{subsec:simloop}

As shown in Fig.~\ref{fig:PIHS},
the proposed physics-informed hybrid search (PIHS)
couples high-level decisions about blocking gaps with
low-level feasibility of multi-robot pushing.
Unlike purely geometric planners,
this procedure simultaneously determines a sequence of gaps to clear
and physically-feasible pushing strategies.
Parallel physics simulation is embedded so that strategies are evaluated simultaneously.
Algorithmic details are summarized in Alg.~\ref{alg:push}
and explained below.


\subsubsection{Tree and Initialization}
The search tree~$\mathcal{T}$ is composed of nodes
$\nu\triangleq(\mathbf{s},\pi)$, where~$\mathbf{s}$ is the current system
state of all robots and movables in~\eqref{eq:transition},
and~$\pi$ the partial pushing strategy in~\eqref{eq:problem}.
Two global functions are
maintained:~$\mathsf{Rank}(\nu)$ stores ranked candidate gaps~$\Gamma_{\mathcal{L}_\nu}$
given the the frontier loop~$\mathcal{L}_\nu$ associated with node~$\nu$,
together with their exploration costs by~\eqref{eq:pred_cost};
and~$\chi(\nu)$ returns the cumulative execution cost
from the root to~$\nu$.
The root is initialized as~$\nu_0\triangleq(\mathbf{s}_0,\emptyset)$,
where~$\mathbf{s}_0$ is the initial system state.
At initialization,~$\chi(\nu_0)=0$, and
$\mathsf{Rank}(\nu_0)$ is constructed by computing~$\Gamma_{\mathcal{L}_{\nu_0}}$
for state~$\mathbf{s}_0$.

\subsubsection{Node Selection}
At each iteration, the node with minimum best-first priority is selected from
the queue. Priority balances realized cost and estimated remaining effort:
\begin{equation}\label{eq:priority}
  f(\nu)\triangleq \chi(\nu)+\underset{g\in \textsf{Rank}(\nu)}{\textbf{min}}
  \widehat{\mathsf{Cost}}(g),
\end{equation}
where~$\chi(\nu)$ is the cumulative execution cost so far, and the second term
is the minimum predicted remaining cost among the unexplored gaps of~$\nu$.
This scoring ensures that nodes are expanded in an order that jointly accounts
for physical effort already incurred and the most promising future actions.



%==============================================
\begin{algorithm}[t!]
\caption{Physics-Informed Hybrid Search}\label{alg:push}
\SetAlgoLined
\KwIn{$\mathbf{s}_0$,~$\alpha$,~$\mathsf{EvalSim}(\cdot)$,~$W$--criterion by~\eqref{eq:wccg_criterion}}
\KwOut{$\pi^\star$}
$\nu_0\gets(\mathbf{s}_0,\emptyset),\;\chi(\nu_0)=0$\;
Init~$\mathcal{Q}$ by~\eqref{eq:priority}\;
\While{not terminated}{
 $\nu\gets\mathcal{Q}.\texttt{pop\_min}()$ by~\eqref{eq:priority}\;
  \ForEach{$g\in\mathsf{Rank}(\nu)$}{
    Compute~$\Pi^\star(g)$, generate~$\Xi_g$\;
    \ForEach{$(\Omega_m,\mathbf{v},\boldsymbol{\xi})\in\Xi_g$}{
      $\tau=(\Omega_m,\mathbf{v},\boldsymbol{\xi})$\tcp{Direct or recur.}
      $(\mathbf{s}',\delta_\texttt{T},\delta_\texttt{J})\gets\mathsf{EvalSim}(\nu,\tau)$
      by~\eqref{eq:sim}\;
    \If{success}{
     $\nu'\gets(\mathbf{s}',\pi\cup\{\tau\})$\;
     $\chi(\nu')\gets\chi(\nu)+\delta_\texttt{T}+\alpha\,\delta_\texttt{J}$\;
     Add $\nu'$ to $\mathcal{Q}$\;
      }
    }
  }
  \If{$\mathbf{s}'$ admits~$\mathcal{P}^W_{\texttt{V}}$ by~\eqref{eq:wccg_criterion}}{
   $\pi^\star\gets\pi'$, \textbf{break}\;
  }
}
\Return~$\pi^\star$;
\end{algorithm}

\subsubsection{Node Expansion with Parallel Simulation}
Let the selected node be~$\nu \triangleq (\mathbf{s},\pi)$. For each ranked gap
$g \in \mathsf{Rank}(\nu)$,
its predicted gap sequence~$\Pi^\star(g)$ in~\eqref{eq:pred_cost}
is first derived. Then, the mode optimization by~\eqref{eq:mode-feasibility}
provides a candidate strategy set
$\Xi_g \triangleq \{(\Omega_m,\mathbf{v},\boldsymbol{\xi})\}$,
including both the direct push of the adjacent movable obstacles,
and recursive push of out-layer obstacles.
Thus, expansion is prioritized first by the gap ordering in
$\mathsf{Rank}(\nu)$, and then by the candidate ordering within~$\Xi_g$.
Each candidate strategy~$(\Omega_m,\mathbf{v},\boldsymbol{\xi}) \in \Xi_g$ is first
checked by a geometric quick-pass and, if it passes, evaluated by the
short-horizon rollout in~\eqref{eq:sim}. A successful evaluation yields
$(\mathbf{s}',\delta_{\texttt{T}},\delta_{\texttt{J}})$, from which a child
node~$\nu' \triangleq (\mathbf{s}',\pi')$ is created by appending
$\tau \triangleq (\Omega_m,\mathbf{v},\boldsymbol{\xi})$ to the current partial
schedule~$\pi$.
The cumulative cost is updated by:
$\chi(\nu') \triangleq \chi(\nu) + \widehat{\mathsf{C}}(\nu,\tau)$,
where~$\widehat{\mathsf{C}}(\nu,\tau)\triangleq
\delta_{\texttt{T}} + \alpha\,\delta_{\texttt{J}}$
is the realized incremental cost.
Note that all candidate strategies in the current batch are evaluated
\emph{concurrently in parallel},
and a single expansion can produce multiple validated successors.




\subsubsection{Termination}
Termination occurs when a node~$\nu$ reaches a state~$\mathbf{s}$ for which a
$W$--clear path~$\mathcal{P}^W_\texttt{V}$ exists from
$\mathbf{s}_\texttt{V}^{\texttt{S}}$ to~$\mathbf{s}_\texttt{V}^{\texttt{G}}$
by~\eqref{eq:wccg_criterion}. Thus, the schedule~$\pi$
stored in~$\nu$ constitutes a complete solution to~\eqref{eq:problem}, encoding
both the sequence of gaps to clear and the pushing strategies
that realize them.

\begin{remark}\label{remark:simloop}
Another efficiency gain above comes from the persistent ModeTable. Effective pushing
modes satisfying~\eqref{eq:mode-feasibility} are stored and later reused for
obstacles with similar local geometry. This reuse avoids repeated contact-mode
optimization and reduces the simulations required
during hybrid search. \hfill$\blacksquare$
\end{remark}
