\subsection{Physics-Informed Hybrid Search}\label{subsec:simloop}

As shown in Fig.~\ref{fig:PIHS}, the proposed physics-informed hybrid search
(PIHS) couples WCCG-guided gap sequencing, recursive push-task generation,
contact-mode optimization, and short-horizon physics rollout. A search node is a
simulated system snapshot. Push tasks label the edges of the search tree: a task
selected at a snapshot is evaluated by physics simulation and produces one or
more successor snapshots. Auxiliary records, including the parent pointer,
the incoming push task, the gap-sequencing cache, the accumulated push effect,
and the accumulated execution cost, are stored only for queue ordering and
solution reconstruction. Thus, WCCG proposes clearance opportunities through
selected portal gaps, while all node transitions are generated and validated by
physics simulation.

\subsubsection{Tree Initialization and Node Value}
The search tree~$\mathcal{T}$ contains snapshot nodes indexed by~$\nu$, each
associated with a full system state~$\mathbf{s}_\nu$ in~\eqref{eq:transition}.
For a non-root node, let~$\nu^-$ be its parent and~$\tau_\nu$ be the incoming
push task that generates~$\mathbf{s}_\nu$ from~$\mathbf{s}_{\nu^-}$ through the
simulation evaluation in~\eqref{eq:sim}. The accepted task sequence is not part
of the node state; it is recovered by tracing the incoming edge labels from a
terminal node back to the root.

For the simulated edge~$\nu^-\xrightarrow{\tau_\nu}\nu$, the accumulated
push-effect score is updated by
\begin{equation}\label{eq:push_effect}
\begin{aligned}
\mathsf{E}(\nu)
&\triangleq
\min\{\mathsf{E}_{\max},
\mathsf{E}(\nu^-)+\Delta\mathsf{E}(\nu^-,\nu)\},\\
\Delta\mathsf{E}(\nu^-,\nu)
&\triangleq
\Delta\mathsf{E}_{\mathrm{obs}}(\mathbf{s}_{\nu^-},\mathbf{s}_{\nu})\\
&\quad
+b_g\mathbb{I}\{d_g^-\le W+\epsilon<d_g\}
+\gamma_g(d_g-d_g^-),
\end{aligned}
\end{equation}
where~$\mathsf{E}_{\max}$ is a finite cap, which can be set to~$\infty$ when
no clipping is used. The term
$\Delta\mathsf{E}_{\mathrm{obs}}(\mathbf{s}_{\nu^-},\mathbf{s}_{\nu})$
measures the weighted change of object poses between the parent and child
snapshots. The quantities~$d_g^-$ and~$d_g$ are the selected gap widths before
and after the rollout, while~$b_g,\gamma_g\ge0$ reward first opening the gap
beyond~$W+\epsilon$ and further enlarging it. For endpoint-clearance tasks,
the gap-width terms in~\eqref{eq:push_effect} are omitted. This score is only a
search heuristic; physical validity is still decided by the rollout checks
in~\eqref{eq:sim}.

The accumulated execution cost along the same branch is updated by
\begin{equation}\label{eq:exec_cost}
\begin{aligned}
\mathsf{C}(\nu)
&\triangleq
\mathsf{C}(\nu^-)+\Delta\mathsf{C}(\nu^-,\nu),\\
\Delta\mathsf{C}(\nu^-,\nu)
&\triangleq
T_{\mathrm{roll}}(\nu^-,\nu)
+\lambda_{\mathrm{tr}}\widehat{C}_{\mathrm{tr}}(\tau_\nu;\mathbf{s}_{\nu^-}),
\end{aligned}
\end{equation}
where~$T_{\mathrm{roll}}$ is the rollout duration or the corresponding number of
control steps, and~$\widehat{C}_{\mathrm{tr}}$ estimates the robot-transfer
cost for reaching the contact configuration of~$\tau_\nu$. The open queue uses
\begin{equation}\label{eq:node_value}
f(\nu)\triangleq
\overline{\mathsf{J}}_{\mathrm{clr}}(\mathbf{s}_\nu)
+(1-\beta)\overline{\mathsf{C}}(\nu)
-\beta\overline{\mathsf{E}}(\nu),
\end{equation}
where~$\overline{\mathsf{J}}_{\mathrm{clr}}$ is the normalized W-clearance
cost-to-go estimate from~\eqref{eq:state_c2g}, while
$\overline{\mathsf{C}}$ and~$\overline{\mathsf{E}}$ are normalized branch cost
and push-effect scores. The parameter~$\beta\in[0,1]$ controls the search
bias: large~$\beta$ favors aggressive feasibility search in dense scenes,
whereas small~$\beta$ gives higher priority to low-cost strategies once
feasible passages are easier to find. Duplicate quantized snapshots are
filtered by keeping the shallowest occurrence.

\subsubsection{Expansion by Lazy Recursive Task Batches}
At each iteration, the node with minimum~\eqref{eq:node_value} is popped and
WCCG-guided gap sequencing is run on its snapshot~$\mathbf{s}_\nu$. If the WCCG
clearance certificate in~\eqref{eq:wccg_criterion} already succeeds, the task
sequence reconstructed from the root to~$\nu$ is returned. If an endpoint is
blocked, endpoint-clearance tasks are sampled by treating the blocked endpoint
as a virtual obstacle. Otherwise, the first portal gap of the best gap sequence
is selected. The task sampler first emits direct pushing tasks for the selected
gap. If these rollouts break the target gap, improve the WCCG clearance
estimate, or exceed the lazy-progress threshold, recursive expansion is skipped.
Recursive blocker-clearing tasks are released only when the direct batch makes
no sufficient progress, or immediately when the same target gap is revisited and
eager direct-plus-recursive sampling is triggered. Each accepted rollout creates
a child snapshot node labeled by the push task that generated it. Successful
children are sorted by~\eqref{eq:node_value}, limited by the per-expansion child
budget, and inserted into an open queue capped by a fixed maximum size.

%==============================================
\begin{algorithm}[t!]
\caption{Physics-Informed Hybrid Search}\label{alg:push}
\SetAlgoLined
\KwIn{$\mathbf{s}_0$, clearance~$W$, child budget~$B$, greediness~$\beta$,
simulator~$\mathsf{EvalSim}$}
\KwOut{push-task sequence~$\pi^\star$}
create root node~$\nu_0$ with~$\mathbf{s}_{\nu_0}\gets\mathbf{s}_0$,
$\mathsf{E}(\nu_0)\gets0$, $\mathsf{C}(\nu_0)\gets0$,
$\mathcal{Q}\gets\{\nu_0\}$\;
\While{$\mathcal{Q}\neq\emptyset$}{
  $\nu\gets\mathcal{Q}.\texttt{pop\_min}()$ by~\eqref{eq:node_value}\;
  $\mathcal{R}_{\nu}\gets\texttt{GapSequencing}(\mathbf{s}_{\nu})$\;
  \If{$\mathbf{s}_{\nu}$ satisfies~\eqref{eq:wccg_criterion}}{
    \Return~$\texttt{TraceTasks}(\nu)$\;
  }
  \eIf{an endpoint is blocked in $\mathcal{R}_{\nu}$}{
    $\mathcal{B}\gets\texttt{SampleEndpointTasks}(\mathbf{s}_{\nu},B)$\;
    $\mathcal{C}\gets\texttt{EvalBatch}(\mathcal{B},\mathbf{s}_{\nu})$\;
  }{
    $g^\star\gets\texttt{FirstGap}(\mathcal{R}_{\nu})$\;
    $\mathcal{B}_{\mathrm{d}}\gets\texttt{SampleDirectTasks}(g^\star,B)$\;
    $\mathcal{C}\gets\texttt{EvalBatch}(\mathcal{B}_{\mathrm{d}},\mathbf{s}_{\nu})$\;
    \If{$\mathcal{C}$ yields no lazy progress}{
      $\mathcal{B}_{\mathrm{r}}\gets\texttt{SampleRecursiveTasks}(g^\star,B)$\;
      $\mathcal{C}\gets\mathcal{C}\cup
      \texttt{EvalBatch}(\mathcal{B}_{\mathrm{r}},\mathbf{s}_{\nu})$\;
    }
  }
  \ForEach{$(\tau,\mathbf{s}',\rho)\in\mathcal{C}$}{
    create child node~$\nu'$ with~$\mathbf{s}_{\nu'}\gets\mathbf{s}'$,
    parent~$\nu$, and incoming task~$\tau$\;
    update~$\mathsf{E}(\nu')$, $\mathsf{C}(\nu')$, and~$f(\nu')$\;
    $\mathcal{Q}.\texttt{push}(\nu')$\;
  }
  trim duplicate snapshots and cap~$\mathcal{Q}$\;
}
\Return~failure\;
\end{algorithm}

\subsubsection{Termination and Replanning}
Termination occurs when a popped node~$\nu$ reaches a snapshot~$\mathbf{s}_\nu$
for which the WCCG clearance certificate in~\eqref{eq:wccg_criterion} succeeds.
The solution~$\pi^\star$ is obtained by tracing the incoming push-task labels
from~$\nu$ to the root, and constitutes a complete solution
to~\eqref{eq:problem}. During execution, the WCCG, the gap-sequencing cache,
and reusable ModeTable entries are updated after each accepted task. If object
drift, transition blockage, or contact loss invalidates the remaining prefix,
the search is restarted from the observed snapshot while preserving failed task
signatures and reusable mode entries.
